Top 10 Nett Sales (Penjualan Bersih) Per Cabang dan Provinsi: 

SELECT 
    kc.branch_id, 
    kc.branch_name, 
    kc.provinsi, 
    ROUND(SUM(ft.nett_sales), 2) AS total_laba_bersih
FROM `Kimia_Farma.rakamin-kf-analytics-4550` AS ft
LEFT JOIN `Kimia_Farma.kf_kantor_cabang` AS kc ON ft.branch_id = kc.branch_id
GROUP BY kc.branch_id, kc.branch_name, kc.provinsi
ORDER BY total_laba_bersih DESC
LIMIT 10;
